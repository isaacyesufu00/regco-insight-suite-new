```sql
-- Agent conversations
CREATE TABLE IF NOT EXISTS public.agent_conversations (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES public.profiles(id) ON DELETE CASCADE NOT NULL,
  title TEXT DEFAULT 'New Conversation',
  preview TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE public.agent_conversations ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users own their conversations" ON public.agent_conversations FOR ALL USING (auth.uid() = user_id);
CREATE INDEX idx_agent_convs_user ON public.agent_conversations(user_id, updated_at DESC);

-- Agent messages
CREATE TABLE IF NOT EXISTS public.agent_messages (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  conversation_id UUID REFERENCES public.agent_conversations(id) ON DELETE CASCADE NOT NULL,
  user_id UUID REFERENCES public.profiles(id) ON DELETE CASCADE NOT NULL,
  role TEXT NOT NULL CHECK (role IN ('user', 'assistant')),
  content TEXT NOT NULL,
  tool_calls JSONB,
  panel_update JSONB,
  actions JSONB,
  created_at TIMESTAMPTZ DEFAULT now()
);
ALTER TABLE public.agent_messages ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users own their messages" ON public.agent_messages FOR ALL USING (auth.uid() = user_id);
CREATE INDEX idx_agent_msgs_conv ON public.agent_messages(conversation_id, created_at ASC);
```

### STEP 4: CREATE src/pages/dashboard/Agent.tsx

The main agent page. Handles conversation loading, message sending, and streaming responses from the agent-orchestrator edge function.

```tsx
import { useState, useEffect, useRef } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import { supabase } from '@/integrations/supabase/client';
import { useAuth } from '@/hooks/useAuth';
import { AgentLayout } from '@/components/agent/AgentLayout';
import { MessageInput } from '@/components/agent/MessageInput';
import { AgentMessage } from '@/components/agent/AgentMessage';
import { motion, AnimatePresence } from 'framer-motion';

const SUGGESTED_PROMPTS = [
  "What returns are due or overdue right now?",
  "Summarise high-risk AML alerts from the last 7 days",
  "Draft an STR narrative for the top flagged transaction",
  "What CBN circulars affected us in the last 30 days?",
  "Generate our latest CBN MFB Regulatory Return",
  "Show me open audit issues grouped by severity",
];

type Message = { id: string; role: 'user' | 'assistant'; content: string; actions?: { label: string; url: string }[]; panelUpdate?: any; created_at: string; };

export default function Agent() {
  const { conversationId } = useParams<{ conversationId?: string }>();
  const { user } = useAuth();
  const navigate = useNavigate();
  const [messages, setMessages] = useState<Message[]>([]);
  const [loading, setLoading] = useState(false);
  const [streamingText, setStreamingText] = useState('');
  const [panelContent, setPanelContent] = useState<React.ReactNode>(null);
  const [panelTitle, setPanelTitle] = useState('');
  const messagesEndRef = useRef<HTMLDivElement>(null);

  // Load conversation messages
  useEffect(() => {
    if (!conversationId || !user) return;
    supabase.from('agent_messages')
      .select('*')
      .eq('conversation_id', conversationId)
      .order('created_at', { ascending: true })
      .then(({ data }) => {
        setMessages(data as Message[] || []);
        setTimeout(() => messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' }), 100);
      });
  }, [conversationId, user]);

  async function sendMessage(text: string) {
    if (!text.trim() || loading || !user) return;
    setLoading(true);
    setStreamingText('');

    // Ensure we have a conversation
    let convId = conversationId;
    if (!convId) {
      const { data } = await supabase.from('agent_conversations')
        .insert({ user_id: user.id, title: text.slice(0, 48) })
        .select().single();
      if (data) {
        convId = data.id;
        navigate(`/dashboard/agent/${data.id}`, { replace: true });
      }
    }
    if (!convId) { setLoading(false); return; }

    // Save user message
    const userMsg: Message = {
      id: crypto.randomUUID(),
      role: 'user',
      content: text,
      created_at: new Date().toISOString(),
    };
    await supabase.from('agent_messages').insert({ ...userMsg, conversation_id: convId, user_id: user.id });
    setMessages(prev => [...prev, userMsg]);
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });

    // Build history for context (last 10 messages)
    const history = [...messages, userMsg].slice(-10).map(m => ({ role: m.role, content: m.content }));

    // Call agent orchestrator
    try {
      const { data: { session } } = await supabase.auth.getSession();
      const response = await supabase.functions.invoke('agent-orchestrator', {
        body: {
          message: text,
          conversation_id: convId,
          history,
          user_id: user.id,
        },
      });

      const result = response.data as { message: string; actions?: { label: string; url: string }[]; panel_update?: any; conversation_title?: string };

      // Update conversation title if first message
      if (result.conversation_title && messages.length === 0) {
        await supabase.from('agent_conversations').update({ title: result.conversation_title, preview: text.slice(0, 60), updated_at: new Date().toISOString() }).eq('id', convId);
      }

      // Save assistant message
      const assistantMsg: Message = {
        id: crypto.randomUUID(),
        role: 'assistant',
        content: result.message,
        actions: result.actions,
        panelUpdate: result.panel_update,
        created_at: new Date().toISOString(),
      };
      await supabase.from('agent_messages').insert({ ...assistantMsg, conversation_id: convId, user_id: user.id, actions: result.actions, panel_update: result.panel_update });
      setMessages(prev => [...prev, assistantMsg]);

      // Handle panel update
      if (result.panel_update) {
        setPanelTitle(result.panel_update.title || 'Context');
        setPanelContent(<pre style={{ fontSize: 11, color: '#4c4c4c', whiteSpace: 'pre-wrap' }}>{JSON.stringify(result.panel_update.data, null, 2)}</pre>);
      }

    } catch (err) {
      const errMsg: Message = {
        id: crypto.randomUUID(), role: 'assistant',
        content: 'I encountered an error processing your request. Please try again.',
        created_at: new Date().toISOString(),
      };
      setMessages(prev => [...prev, errMsg]);
    } finally {
      setLoading(false);
      setTimeout(() => messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' }), 100);
    }
  }

  const isNew = messages.length === 0;

  return (
    <AgentLayout conversationId={conversationId} panelContent={panelContent} panelTitle={panelTitle}>
      <div style={{ display: 'flex', flexDirection: 'column', height: '100%' }}>

        {/* Messages area */}
        <div style={{ flex: 1, overflow: 'auto', padding: '24px 0' }}>
          {isNew ? (
            /* Welcome / suggested prompts */
            <div style={{ maxWidth: 640, margin: '0 auto', padding: '60px 24px 24px', textAlign: 'center' }}>
              <div style={{ width: 44, height: 44, borderRadius: '50%', background: '#0A0A0A', display: 'flex', alignItems: 'center', justifyContent: 'center', margin: '0 auto 20px' }}>
                <span style={{ fontSize: 18, fontWeight: 800, color: '#FFFFFF', fontFamily: 'Bricolage Grotesque, Inter' }}>R</span>
              </div>
              <h2 style={{ fontSize: 22, fontWeight: 700, color: '#0A0A0A', letterSpacing: '-0.5px', margin: '0 0 8px' }}>RegCo Agent</h2>
              <p style={{ fontSize: 14, color: '#6B6B6B', lineHeight: 1.6, margin: '0 0 36px', maxWidth: 400, marginLeft: 'auto', marginRight: 'auto' }}>
                Your compliance copilot. Ask me to generate reports, investigate transactions, draft STRs, or explain your regulatory posture.
              </p>
              <div style={{ display: 'flex', flexDirection: 'column', gap: 8, textAlign: 'left' }}>
                {SUGGESTED_PROMPTS.map(prompt => (
                  <button key={prompt} onClick={() => sendMessage(prompt)}
                    style={{ padding: '11px 16px', background: '#FFFFFF', border: '1px solid rgba(0,0,0,0.08)', borderRadius: 10, cursor: 'pointer', textAlign: 'left', fontSize: 13.5, color: '#0A0A0A', fontWeight: 430, transition: 'background 0.12s, border-color 0.12s', fontFamily: 'Inter, sans-serif' }}
                    onMouseEnter={e => { e.currentTarget.style.background = '#FAFAFA'; e.currentTarget.style.borderColor = 'rgba(0,0,0,0.14)'; }}
                    onMouseLeave={e => { e.currentTarget.style.background = '#FFFFFF'; e.currentTarget.style.borderColor = 'rgba(0,0,0,0.08)'; }}>
                    {prompt}
                  </button>
                ))}
              </div>
            </div>
          ) : (
            <div style={{ maxWidth: 720, margin: '0 auto', padding: '0 24px' }}>
              <AnimatePresence initial={false}>
                {messages.map(msg => (
                  <AgentMessage key={msg.id} message={msg} onActionClick={(url) => navigate(url)} />
                ))}
              </AnimatePresence>
              {loading && (
                <div style={{ display: 'flex', gap: 10, alignItems: 'flex-start', padding: '16px 0' }}>
                  <div style={{ width: 28, height: 28, borderRadius: '50%', background: '#0A0A0A', display: 'flex', alignItems: 'center', justifyContent: 'center', flexShrink: 0 }}>
                    <span style={{ fontSize: 11, fontWeight: 800, color: '#FFF' }}>R</span>
                  </div>
                  <div style={{ background: '#FFFFFF', border: '1px solid rgba(0,0,0,0.07)', borderRadius: '2px 12px 12px 12px', padding: '12px 16px', display: 'flex', gap: 4, alignItems: 'center' }}>
                    {[0,1,2].map(i => (
                      <div key={i} style={{ width: 6, height: 6, borderRadius: '50%', background: '#9B9B9B', animation: `bounce 1.2s ease-in-out ${i*0.2}s infinite` }}/>
                    ))}
                  </div>
                </div>
              )}
              <div ref={messagesEndRef}/>
            </div>
          )}
        </div>

        {/* Input */}
        <div style={{ borderTop: '1px solid rgba(0,0,0,0.07)', padding: '16px 24px', background: '#FAFAFA' }}>
          <div style={{ maxWidth: 720, margin: '0 auto' }}>
            <MessageInput onSend={sendMessage} disabled={loading} />
          </div>
        </div>
      </div>

      {/* Bounce animation */}
      <style>{`@keyframes bounce { 0%,80%,100%{transform:scale(0.7);opacity:0.4} 40%{transform:scale(1);opacity:1} }`}</style>
    </AgentLayout>
  );
}
```

```tsx
import { motion } from 'framer-motion';
import { ExternalLink } from 'lucide-react';

interface MessageProps {
  message: { id: string; role: 'user' | 'assistant'; content: string; actions?: { label: string; url: string }[]; created_at: string; };
  onActionClick: (url: string) => void;
}

export function AgentMessage({ message, onActionClick }: MessageProps) {
  const isUser = message.role === 'user';
  return (
    <motion.div initial={{ opacity: 0, y: 8 }} animate={{ opacity: 1, y: 0 }} transition={{ duration: 0.3, ease: [0.22,1,0.36,1] }}
      style={{ display: 'flex', gap: 10, alignItems: 'flex-start', padding: '10px 0', justifyContent: isUser ? 'flex-end' : 'flex-start' }}>

      {!isUser && (
        <div style={{ width: 28, height: 28, borderRadius: '50%', background: '#0A0A0A', display: 'flex', alignItems: 'center', justifyContent: 'center', flexShrink: 0, marginTop: 2 }}>
          <span style={{ fontSize: 11, fontWeight: 800, color: '#FFF', fontFamily: 'Bricolage Grotesque, Inter' }}>R</span>
        </div>
      )}

      <div style={{ maxWidth: '80%', display: 'flex', flexDirection: 'column', gap: 8 }}>
        <div style={{
          background: isUser ? '#0A0A0A' : '#FFFFFF',
          color: isUser ? '#FFFFFF' : '#0A0A0A',
          borderRadius: isUser ? '12px 2px 12px 12px' : '2px 12px 12px 12px',
          padding: '11px 16px',
          fontSize: 14,
          fontWeight: 400,
          lineHeight: 1.65,
          border: isUser ? 'none' : '1px solid rgba(0,0,0,0.07)',
          whiteSpace: 'pre-wrap',
          fontFamily: 'Inter, sans-serif',
        }}>
          {message.content}
        </div>

        {/* Action buttons */}
        {message.actions && message.actions.length > 0 && (
          <div style={{ display: 'flex', gap: 6, flexWrap: 'wrap' }}>
            {message.actions.map(action => (
              <button key={action.label} onClick={() => onActionClick(action.url)}
                style={{ fontFamily: 'Inter, sans-serif', fontSize: 12, fontWeight: 600, color: '#0A0A0A', background: '#F5F5F0', border: '1px solid rgba(0,0,0,0.09)', padding: '5px 11px', borderRadius: 7, cursor: 'pointer', display: 'flex', alignItems: 'center', gap: 4, transition: 'background 0.12s' }}
                onMouseEnter={e => { e.currentTarget.style.background = '#EEEEEA'; }}
                onMouseLeave={e => { e.currentTarget.style.background = '#F5F5F0'; }}>
                <ExternalLink size={10}/> {action.label}
              </button>
            ))}
          </div>
        )}

        <div style={{ fontSize: 10.5, color: '#9B9B9B', paddingLeft: isUser ? 0 : 4 }}>
          {new Date(message.created_at).toLocaleTimeString('en-NG', { hour: '2-digit', minute: '2-digit' })}
        </div>
      </div>
    </motion.div>
  );
}
```

```tsx
import { motion, AnimatePresence } from 'framer-motion';

const TOOLS = [
  { group: 'Files', items: [{ icon: Upload, label: 'Upload CBS File', command: 'Upload CBS: ' }, { icon: FileText, label: 'Upload Transactions CSV', command: 'Analyze transactions from: ' }] },
  { group: 'Actions', items: [{ icon: BarChart2, label: 'Generate Report', command: 'Generate report: ' }, { icon: User, label: 'Customer 360', command: 'Show full profile for customer: ' }, { icon: Shield, label: 'Screen Customer', command: 'Screen customer for sanctions: ' }] },
  { group: 'Quick Views', items: [{ icon: Newspaper, label: 'Latest CBN Circulars', command: 'Summarise recent CBN circulars' }, { icon: BarChart2, label: 'Compliance Status', command: 'What returns are due or overdue right now?' }] },
];

interface MessageInputProps {
  onSend: (text: string) => void;
  disabled: boolean;
}

export function MessageInput({ onSend, disabled }: MessageInputProps) {
  const [text, setText] = useState('');
  const [toolsOpen, setToolsOpen] = useState(false);
  const textareaRef = useRef<HTMLTextAreaElement>(null);

  function handleKeyDown(e: React.KeyboardEvent) {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      submit();
    }
  }

  function submit() {
    if (!text.trim() || disabled) return;
    onSend(text.trim());
    setText('');
    setToolsOpen(false);
  }

  function selectTool(command: string) {
    setText(command);
    setToolsOpen(false);
    textareaRef.current?.focus();
  }

  return (
    <div style={{ position: 'relative' }}>
      {/* Tool selector popover */}
      <AnimatePresence>
        {toolsOpen && (
          <motion.div initial={{ opacity: 0, y: 8 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0, y: 8 }} transition={{ duration: 0.18 }}
            style={{ position: 'absolute', bottom: '100%', left: 0, marginBottom: 8, background: '#FFFFFF', border: '1px solid rgba(0,0,0,0.09)', borderRadius: 12, boxShadow: '0 8px 32px rgba(0,0,0,0.12)', padding: '10px', width: 280, zIndex: 50 }}>
            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginBottom: 10, paddingBottom: 8, borderBottom: '1px solid rgba(0,0,0,0.06)' }}>
              <span style={{ fontSize: 11, fontWeight: 700, color: '#0A0A0A', textTransform: 'uppercase', letterSpacing: '0.07em' }}>Tools</span>
              <button onClick={() => setToolsOpen(false)} style={{ background: 'none', border: 'none', cursor: 'pointer', color: '#9B9B9B', padding: 2, borderRadius: 4 }}><X size={12}/></button>
            </div>
            {TOOLS.map(group => (
              <div key={group.group} style={{ marginBottom: 10 }}>
                <p style={{ fontSize: 10, fontWeight: 600, color: '#9B9B9B', textTransform: 'uppercase', letterSpacing: '0.07em', margin: '0 0 5px', paddingLeft: 4 }}>{group.group}</p>
                {group.items.map(item => {
                  const Icon = item.icon;
                  return (
                    <button key={item.label} onClick={() => selectTool(item.command)}
                      style={{ width: '100%', padding: '8px 8px', background: 'none', border: 'none', borderRadius: 7, cursor: 'pointer', display: 'flex', alignItems: 'center', gap: 9, textAlign: 'left', fontSize: 13, fontWeight: 450, color: '#0A0A0A', transition: 'background 0.1s', fontFamily: 'Inter, sans-serif' }}
                      onMouseEnter={e => { e.currentTarget.style.background = '#F5F5F0'; }}
                      onMouseLeave={e => { e.currentTarget.style.background = 'none'; }}>
                      <Icon size={13} style={{ color: '#6B6B6B', flexShrink: 0 }}/>
                      {item.label}
                    </button>
                  );
                })}
              </div>
            ))}
          </motion.div>
        )}
      </AnimatePresence>

      {/* Input row */}
      <div style={{ display: 'flex', gap: 8, alignItems: 'flex-end', background: '#FFFFFF', border: '1px solid rgba(0,0,0,0.10)', borderRadius: 12, padding: '10px 10px 10px 12px', boxShadow: '0 1px 4px rgba(0,0,0,0.05)' }}>
        {/* Plus button */}
        <button onClick={() => setToolsOpen(s => !s)}
          style={{ width: 30, height: 30, flexShrink: 0, background: toolsOpen ? '#0A0A0A' : '#F5F5F0', border: 'none', borderRadius: 7, cursor: 'pointer', display: 'flex', alignItems: 'center', justifyContent: 'center', transition: 'background 0.15s', color: toolsOpen ? '#FFFFFF' : '#6B6B6B' }}>
          <Plus size={14}/>
        </button>

        {/* Textarea */}
        <textarea ref={textareaRef} value={text} onChange={e => setText(e.target.value)} onKeyDown={handleKeyDown}
          placeholder="Ask RegCo Agent to generate reports, review alerts, draft STRs, or explain your risk posture…"
          disabled={disabled}
          rows={1}
          style={{ flex: 1, background: 'none', border: 'none', outline: 'none', resize: 'none', fontSize: 14, color: '#0A0A0A', fontFamily: 'Inter, -apple-system, sans-serif', lineHeight: 1.55, minHeight: 22, maxHeight: 120, overflow: 'auto' }}
          onInput={e => { const t = e.currentTarget; t.style.height = 'auto'; t.style.height = Math.min(t.scrollHeight, 120) + 'px'; }}
        />

        {/* Send button */}
        <button onClick={submit} disabled={!text.trim() || disabled}
          style={{ width: 30, height: 30, flexShrink: 0, background: text.trim() && !disabled ? '#0A0A0A' : '#F0EDE8', border: 'none', borderRadius: 7, cursor: text.trim() && !disabled ? 'pointer' : 'default', display: 'flex', alignItems: 'center', justifyContent: 'center', transition: 'background 0.15s' }}>
          <Send size={13} style={{ color: text.trim() && !disabled ? '#FFFFFF' : '#9B9B9B' }}/>
        </button>
      </div>

      <p style={{ fontSize: 10.5, color: '#9B9B9B', textAlign: 'center', marginTop: 6 }}>RegCo Agent may make mistakes. Always verify before filing with regulators.</p>
    </div>
  );
}
```
